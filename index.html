const makeWASocket = require('@whiskeysockets/baileys').default; const { useMultiFileAuthState, DisconnectReason } = require('@whiskeysockets/baileys'); const Ably = require('ably'); const http = require('http'); // Initialize Ably connection using the environment variable we will set in Zeabur const ABLY_API_KEY = process.env.ABLY_API_KEY; const ably = new Ably.Rest(ABLY_API_KEY); const channel = ably.channels.get('whatsapp'); async function connectToWhatsApp() { // Persist login state to local disk storage folder const { state, saveCreds } = await useMultiFileAuthState('auth_info_baileys'); const sock = makeWASocket({ auth: state, printQRInTerminal: true // Automatically outputs a clean text QR code in Zeabur logs }); sock.ev.on('creds.update', saveCreds); sock.ev.on('connection.update', (update) => { const { connection, lastDisconnect } = update; if(connection === 'close') { const shouldReconnect = lastDisconnect.error?.output?.statusCode !== DisconnectReason.loggedOut; console.log('Connection closed, reconnecting: ', shouldReconnect); if(shouldReconnect) connectToWhatsApp(); } else if(connection === 'open') { console.log('🚀 SUCCESS: WhatsApp Cloud Worker is fully connected!'); } }); // Catch inbound messages and push straight to Ably sock.ev.on('messages.upsert', async m => { if (m.type === 'notify') { for (const msg of m.messages) { if (!msg.key.fromMe && msg.message?.conversation) { const messagePayload = { text: msg.message.conversation, sender: msg.key.remoteJid }; channel.publish('msg', JSON.stringify(messagePayload)); } } } }); } connectToWhatsApp(); // Standard port binding to tell Zeabur the web service is healthy http.createServer((req, res) => res.end("Live")).listen(process.env.PORT || 3000);
