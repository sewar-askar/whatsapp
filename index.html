const makeWASocket = require('@whiskeysockets/baileys').default; const { useMultiFileAuthState, DisconnectReason } = require('@whiskeysockets/baileys'); const Ably = require('ably'); // 1. Initialize Ably Connection const ABLY_API_KEY = process.env.ABLY_API_KEY; const ably = new Ably.Rest(ABLY_API_KEY); const channel = ably.channels.get('whatsapp'); async function connectToWhatsApp() { // Persist login state to local disk storage folder const { state, saveCreds } = await useMultiFileAuthState('auth_info_baileys'); const sock = makeWASocket({ auth: state, printQRInTerminal: true // Generates text-based QR code in your dashboard logs }); sock.ev.on('creds.update', saveCreds); // Monitor connectivity events sock.ev.on('connection.update', (update) => { const { connection, lastDisconnect } = update; if(connection === 'close') { const shouldReconnect = lastDisconnect.error?.output?.statusCode !== DisconnectReason.loggedOut; console.log('Connection closed due to ', lastDisconnect.error, ', reconnecting: ', shouldReconnect); if(shouldReconnect) connectToWhatsApp(); } else if(connection === 'open') { console.log('🚀 SUCCESS: WhatsApp Cloud Worker is fully connected!'); } }); // Ingest incoming texts and immediately route to Ably sock.ev.on('messages.upsert', async m => { if (m.type === 'notify') { for (const msg of m.messages) { if (!msg.key.fromMe && msg.message?.conversation) { const messagePayload = { text: msg.message.conversation, sender: msg.key.remoteJid }; // Fire-and-forget payload publication to Ably channel.publish('msg', JSON.stringify(messagePayload)); } } } }); } connectToWhatsApp(); // Simple Express-like dummy server binding to keep the free container listening active const http = require('http'); http.createServer((req, res) => res.end("Bridge running.")).listen(process.env.PORT || 3000);
